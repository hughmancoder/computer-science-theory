//selection for x
   Mux16(a = x, sel = zx, out = x1); // Zero x (if sel == 1, out = b)
   Not16(in = x1, out = negx); // Not x
   Mux16(a = x1, b = negx, sel = nx, out = x2); //chose x1 or negate x1

   //selection for y
   Mux16(a = y, sel = zy, out = y1); // Zero y (if sel == 1, out = b)
   Not16(in = y1, out = negy); // Not y
   Mux16(a = y1, b = negy, sel = ny, out = y2); //chose x1 or negate x1

   //x + y or x & y 
   Add16(a=x2,b=y2,out=xaddy);
   And16(a=x2,b=y2,out=xandy);
   Mux16(a=xandy,b=xaddy,sel=f,out=res);

   //negate the output
   Not16(in=res,out=negres);
   Mux16(a=res,b=negres,sel=no,out=out);
